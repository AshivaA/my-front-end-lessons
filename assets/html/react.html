2-setInterval() timer using useEffect example :
The setInterval() method executes a function repeatedly at a specified interval(time space).

import React, { useState, useEffect } from "react";

const Timer = () => {
const [count, setCount] = useState(0);

useEffect(() => {
const interval = setInterval(() => {
setCount(count + 1);
}, 1000);

return () => clearInterval(interval);

}, [count]);

return (
< div>
< h1>{count} second passed< /h1>
< /div>
);
};

export default Timer;
--
3-setTimeout() timer using useEffect example :
The setTimeout method calls a function or runs some code after a period of time.

import React, { useState, useEffect } from "react";

const Timer = () => {
const [count, setCount] = useState(0);

useEffect(() => {
const Timer = setTimeout(() => {
setCount("this sentence will appear after 3 seconds");
}, 3000);

return () => clearTimeout(Timer);
}, [count]);

return (
< div>
< h1>{count}< /h1>
< /div>
);
};

export default Timer;
